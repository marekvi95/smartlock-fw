426 static inline void I2C_MasterClearStatusFlags(I2C_Type *base, uint32_t statusMask)
427 {
428 /* Must clear the STARTF / STOPF bits prior to clearing IICIF */
429 #if defined(FSL_FEATURE_I2C_HAS_START_STOP_DETECT) && FSL_FEATURE_I2C_HAS_START_STOP_DETECT
430  if ((uint32_t)kI2C_StartDetectFlag == (statusMask & (uint32_t)kI2C_StartDetectFlag))
431  {
432  /* Shift the odd-ball flags back into place. */
433  base->FLT |= (uint8_t)(statusMask >> 8U);
434  }
435 #endif
436 
437 #ifdef I2C_HAS_STOP_DETECT
438  if ((uint32_t)kI2C_StopDetectFlag == (statusMask & (uint32_t)kI2C_StopDetectFlag))
439  {
440  /* Shift the odd-ball flags back into place. */
441  base->FLT |= (uint8_t)(statusMask >> 8U);
442  }
443 #endif
444 
445  base->S = (uint8_t)statusMask;
446 }
