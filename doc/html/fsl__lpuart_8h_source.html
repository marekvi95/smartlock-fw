600 static inline uint8_t LPUART_ReadByte(LPUART_Type *base)
601 {
602 #if defined(FSL_FEATURE_LPUART_HAS_7BIT_DATA_SUPPORT) && FSL_FEATURE_LPUART_HAS_7BIT_DATA_SUPPORT
603  uint32_t ctrl = base->CTRL;
604  uint8_t result;
605  bool isSevenDataBits =
606  (((ctrl & LPUART_CTRL_M7_MASK) ? true : false) ||
607  (((!(ctrl & LPUART_CTRL_M7_MASK)) ? true : false) && ((!(ctrl & LPUART_CTRL_M_MASK)) ? true : false) &&
608  ((ctrl & LPUART_CTRL_PE_MASK) ? true : false)));
609 
610  if (isSevenDataBits)
611  {
612  result = (uint8_t)(base->DATA & 0x7FU);
613  }
614  else
615  {
616  result = (uint8_t)base->DATA;
617  }
618 
619  return result;
620 #else
621  return (uint8_t)(base->DATA);
622 #endif
623 }
