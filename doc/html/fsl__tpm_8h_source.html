491 static inline uint32_t TPM_GetStatusFlags(TPM_Type *base)
492 {
493  uint32_t statusFlags = 0;
494 
495 #if defined(FSL_FEATURE_TPM_HAS_NO_STATUS) && FSL_FEATURE_TPM_HAS_NO_STATUS
496  uint8_t chanlNumber = 0;
497 
498  /* Check the timer flag */
499  if (base->SC & TPM_SC_TOF_MASK)
500  {
501  statusFlags |= kTPM_TimeOverflowFlag;
502  }
503 
504  for (chanlNumber = 0; chanlNumber < FSL_FEATURE_TPM_CHANNEL_COUNTn(base); chanlNumber++)
505  {
506  /* Check the channel flag */
507  if (base->CONTROLS[chanlNumber].CnSC & TPM_CnSC_CHF_MASK)
508  {
509  statusFlags |= (1U << chanlNumber);
510  }
511  }
512 #else
513  statusFlags = base->STATUS;
514 #endif
515 
516  return statusFlags;
517 }

526 static inline void TPM_ClearStatusFlags(TPM_Type *base, uint32_t mask)
527 {
528 #if defined(FSL_FEATURE_TPM_HAS_NO_STATUS) && FSL_FEATURE_TPM_HAS_NO_STATUS
529  uint32_t chnlStatusFlags = (mask & 0xFF);
530  uint8_t chnlNumber = 0;
531 
532  /* Clear the timer overflow flag by writing a 0 to the bit while it is set */
533  if (mask & kTPM_TimeOverflowFlag)
534  {
535  base->SC &= ~TPM_SC_TOF_MASK;
536  }
537  /* Clear the channel flag */
538  while (chnlStatusFlags)
539  {
540  if (chnlStatusFlags & 0x1)
541  {
542  base->CONTROLS[chnlNumber].CnSC &= ~TPM_CnSC_CHF_MASK;
543  }
544  chnlNumber++;
545  chnlStatusFlags = chnlStatusFlags >> 1U;
546  }
547 #else
548  /* Clear the status flags */
549  base->STATUS = mask;
550 #endif
551 }

627 {
628 #if defined(FSL_FEATURE_TPM_HAS_SC_CLKS) && FSL_FEATURE_TPM_HAS_SC_CLKS
629  /* Set clock source to none to disable counter */
630  base->SC &= ~(TPM_SC_CLKS_MASK);
631

633  while (base->SC & TPM_SC_CLKS_MASK)

638  base->SC &= ~(TPM_SC_CMOD_MASK);
639
